    // –ù–∞–π–¥–∏—Ç–µ –∑–∞—â–∏—â—ë–Ω–Ω–æ–µ —Å–æ–¥–µ—Ä–∂–∏–º–æ–µ
    const protectedContent = document.querySelector(".protected-content");

    if (userPassword === correctPassword) {
        // –ï—Å–ª–∏ –ø–∞—Ä–æ–ª—å –≤–µ—Ä–Ω—ã–π, –ø–æ–∫–∞–∂–∏—Ç–µ —Å–æ–¥–µ—Ä–∂–∏–º–æ–µ
        if (protectedContent) {
            protectedContent.style.display = "block"; // –ü–æ–∫–∞–∑—ã–≤–∞–µ–º —Å–æ–¥–µ—Ä–∂–∏–º–æ–µ
        }
    } else {
        // –ï—Å–ª–∏ –ø–∞—Ä–æ–ª—å –Ω–µ–≤–µ—Ä–Ω—ã–π, —Å–∫—Ä–æ–π—Ç–µ —Å–æ–¥–µ—Ä–∂–∏–º–æ–µ –∏ –ø–æ–∫–∞–∂–∏—Ç–µ —Å–æ–æ–±—â–µ–Ω–∏–µ
        if (protectedContent) {
            protectedContent.style.display = "none"; // –°–∫—Ä—ã–≤–∞–µ–º —Å–æ–¥–µ—Ä–∂–∏–º–æ–µ
        }
        document.body.innerHTML = "<h1>–î–æ—Å—Ç—É–ø –∑–∞–ø—Ä–µ—â–µ–Ω</h1><p>–ù–µ–≤–µ—Ä–Ω—ã–π –ø–∞—Ä–æ–ª—å.</p>";
    }

    // –§—É–Ω–∫—Ü–∏—è –¥–ª—è –Ω–æ—Ä–º–∞–ª–∏–∑–∞—Ü–∏–∏ –ø–∞—Ä–æ–ª—è
    function normalizePassword(password) {
        // 1. –£–¥–∞–ª—è–µ–º –ª–∏—à–Ω–∏–µ –ø—Ä–æ–±–µ–ª—ã –≤ –Ω–∞—á–∞–ª–µ –∏ –∫–æ–Ω—Ü–µ
        password = password.trim();

        // 2. –ü—Ä–∏–≤–æ–¥–∏–º –∫ –Ω–∏–∂–Ω–µ–º—É —Ä–µ–≥–∏—Å—Ç—Ä—É
        password = password.toLowerCase();

        // 3. –ó–∞–º–µ–Ω—è–µ–º —Å–∏–º–≤–æ–ª—ã, –≤–≤–µ–¥—ë–Ω–Ω—ã–µ –≤ —Ä—É—Å—Å–∫–æ–π —Ä–∞—Å–∫–ª–∞–¥–∫–µ, –Ω–∞ –∞–Ω–≥–ª–∏–π—Å–∫–∏–µ
        const ruToEnMap = {
            "—Å": "s",
            "–∏": "i",
            "–≤": "v",
            "–∞": "a",
            "–∫": "k",
            "–æ": "o",
            "–≤": "v"
        };

        // –ü—Ä–æ—Ö–æ–¥–∏–º –ø–æ –∫–∞–∂–¥–æ–º—É —Å–∏–º–≤–æ–ª—É –∏ –∑–∞–º–µ–Ω—è–µ–º —Ä—É—Å—Å–∫–∏–µ –±—É–∫–≤—ã –Ω–∞ –∞–Ω–≥–ª–∏–π—Å–∫–∏–µ
        password = password.replace(/./g, char => ruToEnMap[char] || char);

        return password;
    }
});
